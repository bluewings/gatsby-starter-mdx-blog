## Context 1

<Context>
  {({ value, setValue }) => {
    const handleClick = () => setValue('count', (value.count || 0) + 1);
    return <button type="button" onClick={handleClick}>add count</button>
  }}
</Context>

<Context>
  {({ value }) => <pre>{JSON.stringify(value)}</pre>}
</Context>
